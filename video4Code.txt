#include <Joystick.h>

Joystick_ Joystick;
int rX = 0;
int rY = 0;
int lX = 0;
int lY = 0;

bool A, B, X, Y;
const bool initAutoSendState = true;

void setup() {
  Serial.begin(9600);
  
  pinMode(7, INPUT_PULLUP); // Left Shoulder button
  pinMode(6, INPUT_PULLUP); // Right Shoulder button
  pinMode(9,INPUT_PULLUP); // Left Button 1
  pinMode(10,INPUT_PULLUP);// Left Button 2
  pinMode(11,INPUT_PULLUP);// Right Button 1
  pinMode(12,INPUT_PULLUP); // Right Button 2
  
  pinMode(LED_BUILTIN, OUTPUT); // Power LED

  Joystick.begin();

}

void loop() {

  rY = analogRead(A3); 
  Joystick.setRyAxis(rX);
  
  lX = analogRead(A0); 
  Joystick.setXAxis(lX);

  lY = analogRead(A1);
  lY = map(lY, 0, 1023, 1023, 0);
  Joystick.setYAxis(lY);

 if (digitalRead(9) == LOW)

    Joystick.releaseButton(2);

 {

 else
